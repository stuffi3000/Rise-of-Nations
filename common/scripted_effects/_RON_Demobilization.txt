RON_handle_post_ww1_demob = {
	### Force Demobilization After WW1
	if = {
		limit = {
			NOT = { has_tech = demobilization_machine_tools }
		}
		add_ideas = RON_must_demobilize_ww1_a
	}
	if = {
		limit = {
			NOT = { has_tech = demobilization2 }
		}
		add_ideas = RON_must_demobilize_ww1_b
	}
	if = {
		limit = {
			NOT = { has_tech = constructiondemob }
		}
		add_ideas = RON_must_demobilize_ww1_c
	}
	#set_country_flag = RON_Needs_To_Demobilize_WW1
}

RON_handle_post_ww2_demob = {
	if = {
		limit = {
			NOT = { has_tech = demobilization_ww2_a }
			NOT = { has_tech = demobilization_ww2_b }
		}
		add_ideas = RON_must_demobilize_ww2_a
	}
	if = {
		limit = {
			NOT = { has_tech = demobilization_ww2_c }
			NOT = { has_tech = demobilization_ww2_d }
		}
		add_ideas = RON_must_demobilize_ww2_b
	}
	if = {
		limit = {
			NOT = { has_tech = demobilization_ww2_e }
		}
		add_ideas = RON_must_demobilize_ww2_c
	}
	if = {
		limit = {
			NOT = { has_tech = demobilization_ww2_f }
		}
		add_ideas = RON_must_demobilize_ww2_d
	}
	if = {
		limit = {
			NOT = { has_tech = demobilization_ww2_g }
		}
		add_ideas = RON_must_demobilize_ww2_e
	}
	#set_country_flag = RON_Needs_To_Demobilize_WW2
}
