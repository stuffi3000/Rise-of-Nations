X_AUS_pick_propaganda_campaign_slot_effect = {
	hidden_effect = {
		if = {
			limit = {
				check_variable = { X_AUS_poster_left_icon_frame = X_AUS_propaganda_empty_slot_index }
			}
			set_variable = { X_AUS_poster_left_icon_frame = X_AUS_new_propaganda_campaign }
			#Ageing posters
			set_variable = { X_AUS_propaganda_left_poster_to_age = X_AUS_poster_left_icon_frame } #This is used to prevent the system to break if a propaganda campaign is cancelled
			hidden_effect = {
				country_event = { id = xpand_AUS_capmaign_events.101 days = X_AUS_default_propaganda_campaign_old_days }
				#set_country_flag = { flag = X_AUS_propaganda_left_days_remaining_flag days = X_AUS_default_propaganda_campaign_duration } #Surprise! This doesnät work with vars :D  Maybe in the future
				if = { #Debug -> Faster time
					limit = {
						has_country_flag = X_AUS_debug_fast_propaganda_campaigns_flag
					}
					set_country_flag = { flag = X_AUS_propaganda_left_days_remaining_flag days = 5 }
				}
				else = { #Regular time
					set_country_flag = { flag = X_AUS_propaganda_left_days_remaining_flag days = 180 } #Need code support to use a var here
				}
			}
		}

				}
				else = { #Regular time
					set_country_flag = { flag = X_AUS_propaganda_center_days_remaining_flag days = 180 } #Need code support to use a var here

X_AUS_reset_propaganda_campaign_slot_effect = {
	hidden_effect = {
		if = {
			limit = {
				check_variable = { X_AUS_poster_left_icon_frame = X_AUS_propaganda_campaign_to_remove }
			}
			set_variable = { X_AUS_poster_left_icon_frame = X_AUS_propaganda_empty_slot_index }
			#Clear aged posters
			clear_variable = X_AUS_propaganda_left_poster_to_age
			clr_country_flag = X_AUS_propaganda_left_slot_is_old_flag
		}
		else_if = {
			limit = {
				check_variable = { X_AUS_poster_center_icon_frame = X_AUS_propaganda_campaign_to_remove }
			}
			set_variable = { X_AUS_poster_center_icon_frame = X_AUS_propaganda_empty_slot_index }
			#Clear aged posters
			clear_variable = X_AUS_propaganda_center_poster_to_age
			clr_country_flag = X_AUS_propaganda_center_slot_is_old_flag
		}
		else_if = { #This could be turned into an else, but leaving it like this to hopefully make it easier to see
			limit = {
				check_variable = { X_AUS_poster_right_icon_frame = X_AUS_propaganda_campaign_to_remove }
			}
			set_variable = { X_AUS_poster_right_icon_frame = X_AUS_propaganda_empty_slot_index }
			#Clear aged posters
			clear_variable = X_AUS_propaganda_right_poster_to_age
			clr_country_flag = X_AUS_propaganda_right_slot_is_old_flag
		}
		set_variable = { X_AUS_propaganda_campaign_to_remove = 0 } # Reset var
	}
}
