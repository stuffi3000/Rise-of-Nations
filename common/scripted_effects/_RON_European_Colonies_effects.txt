#Give Event to Spain
RON_partition_usa_with_spain = {
	if = {
		limit = {
			NOT = { is_subject_of = POR }
			NOT = { is_subject_of = AUS }
			NOT = { is_subject_of = AUH }
			country_exists = SPR
		}
		SPR = {
			country_event = {
				id = DNC_Spain.14
				hours = 1
			}
		}
	}
	if = {
		limit = {
			country_exists = POR
			OR = {
				SPR = { is_subject_of = ROOT }
				all_state = { if = { limit = { is_core_of = SPR } is_owned_by = ROOT } }
			}
		}
		POR = {
			country_event = {
				id = DNC_Spain.14
				hours = 1
			}
		}
	}
	if = {
		limit = {
			country_exists = AUH
			OR = {
				SPR = { is_subject_of = ROOT }
				all_state = { if = { limit = { is_core_of = SPR } is_owned_by = ROOT } }
			}
		}
		AUH = {
			country_event = {
				id = DNC_Spain.14
				hours = 1
			}
		}
	}
	if = {
		limit = {
			country_exists = AUS
			OR = {
				SPR = { is_subject_of = ROOT }
				all_state = { if = { limit = { is_core_of = SPR } is_owned_by = ROOT } }
			}
		}
		AUS = {
			country_event = {
				id = DNC_Spain.14
				hours = 1
			}
		}
	}
}
