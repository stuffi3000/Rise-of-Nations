army_desertion = {

	random_country = {
		limit = {
			AND = {
				country_exists = THIS
				is_ai = yes

				OR = {
					tag = MCO
					tag = ADR
					tag = LIE
					tag = DEL
					tag = VCT
					tag = LUX
					tag = LON
					tag = SMR
					tag = TGR
				}
			}
		}
		ROOT = {
			if = {
				limit = {
					has_idea = famine_0
				}
				transfer_units_fraction = {
					target = PREV
					size = 0.05
					stockpile_ratio = 0.05
					army_ratio = 0.05
					navy_ratio = 0.05
					air_ratio = 0.05


				}
				if = {
					limit = {
						has_global_flag = test_features
					}
				}
				log = "[GetYear] [GetMonth] | DEBUG | [ROOT.GetName] | lost 5% of his divisions"
			}
			else_if = {
				limit = {
					has_idea = famine_1

				}
				transfer_units_fraction = {
					target = PREV
					size = 0.1
					stockpile_ratio = 0.1
					army_ratio = 0.1
					navy_ratio = 0.1
					air_ratio = 0.1
				}
				if = {
					limit = {
						has_global_flag = test_features
					}
				}
				log = "[GetYear] [GetMonth] | DEBUG | [ROOT.GetName] | lost 10% of his divisions"
			}
			else_if = {
				limit = {
					has_idea = famine_2
				}
				transfer_units_fraction = {
					target = PREV
					size = 0.2
					stockpile_ratio = 0.2
					army_ratio = 0.2
					navy_ratio = 0.2
					air_ratio = 0.2
				}
				if = {
					limit = {
						has_global_flag = test_features
					}
				}
				log = "[GetYear] [GetMonth] | DEBUG | [ROOT.GetName] | lost 20% of his divisions"
			}
			else_if = {
				limit = {
					has_idea = famine_3
				}
				transfer_units_fraction = {
					target = PREV
					size = 0.5
					stockpile_ratio = 0.5
					army_ratio = 0.5
					navy_ratio = 0.5
					air_ratio = 0.5
				}
				if = {
					limit = {
						has_global_flag = test_features
					}
				}
				log = "[GetYear] [GetMonth] | DEBUG | [ROOT.GetName] | lost 50% of his divisions"
				if = {
					limit = {
						tag = GER
						OR = {
							has_war_with = FRA
							has_war_with = RUS
							has_war_with = ENG
						}
					}
					country_event = { id = ww1_germany.58 }
				}
				else_if = {
					limit = {
						tag = AUH
						has_war = yes
					}
					country_event = { id = ww1_kuk.51 }
				}
				else_if = {
					limit = {
						tag = FRA
						has_war_with = GER
					}
					GER = {
						country_event = { id = kaiserschlachtvictory.4 }
					}
				}
				else_if = {
					limit = {
						has_war = yes
					}
					if = {
						limit = {
							has_idea = RI_surrender
						}
						add_timed_idea = {
							idea = RI_surrender_visible
							days = 30
						}
						hidden_effect = {
							modify_timed_idea = {
								idea = RI_surrender
								days = 30
							}
						}
					}
					else = {
						add_timed_idea = {
							idea = RI_surrender_visible
							days = 30
						}
						hidden_effect = {
							add_timed_idea = {
								idea = RI_surrender
								days = 30
							}
						}
					}
				}
			}
		}
		random_country = {
			limit = {
				NOT = {
					has_country_flag = army_dump
				}
			}
			set_country_flag = army_desertion
			annex_country = { target = PREV transfer_troops = no }
			release = PREV
			clr_country_flag = army_desertion
		}

	}
}
