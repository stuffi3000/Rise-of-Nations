on_actions = {
	on_capitulation = {
		effect = {

			if = {
				limit = {
					ROOT = {
						tag = DNZ
					}
				}
				FROM = { annex_country = { target = DNZ } }
			}

			if = {
				limit = {
					ROOT = {
						original_tag = GER
						has_government = fascism
						has_civil_war = no
					}
				}
				GER = {
					country_event = { id = bs.12 }
				}
			}

		}
	}

	on_uncapitulation = {
		effect = {
			if = {
				limit = {
					original_tag = GER
					has_government = fascism
					is_puppet = no
				}
				drop_cosmetic_tag = yes
			}
		}
	}

	on_declare_war = {
		effect = {
			if = {
				limit = {
					ROOT = {
						original_tag = GER
					}
					FROM = {
						tag = DNZ
					}
				}
				DNZ = {
					country_event = { hours = 0 id = danzig.10 }
				}
			}
		}
	}

	on_state_control_changed = {
		effect = {
			if = {
				limit = {
					FROM.FROM = { state = 889 }
					NOT = { has_global_flag = occupied_schutzzone }
					ROOT = {
						tag = SLO
						is_puppet_of = GER
						has_government = fascism
					}
				}
				set_global_flag = schutzzone_treaty
			}
		}
	}

	on_startup = {
		effect = {
			GER = {
				set_capital = {
					state = 1105
				}
			}
		}
	}

	on_puppet = {
		effect = {
			if = {
				limit = {
					AND = {
						TAG = GRE
						is_subject_of = GER
						GER = { has_government = fascism }
					}
				}
				set_cosmetic_tag = GRE_GER

				create_country_leader = {
					name = "Konstantinos Logothetopoulos"
					desc = "POLITICS_KONSTANTINOS_LOGOTHETOPOULOS_DESC"
					picture = "Portrait_Greece_Konstantinos_Logothetopoulos.dds"
					expire = "1965.1.1"
					ideology = fascism_ideology
					traits = {
						great_war_hero
					}
				}
			}

		}
	}

	on_release_as_puppet = {
		effect = {
			if = {
				limit = {
					AND = {
						TAG = GRE
						is_subject_of = GER
						GER = { has_government = fascism }
					}
				}
				set_cosmetic_tag = GRE_GER

				create_country_leader = {
					name = "Konstantinos Logothetopoulos"
					desc = "POLITICS_KONSTANTINOS_LOGOTHETOPOULOS_DESC"
					picture = "Portrait_Greece_Konstantinos_Logothetopoulos.dds"
					expire = "1965.1.1"
					ideology = fascism_ideology
					traits = {
						great_war_hero
					}
				}
			}
		}
	}
}
