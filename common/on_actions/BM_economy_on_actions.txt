on_actions = {
	on_startup = {
		effect = {
			every_country = {
				if = {
					limit = {
						NOT = { is_in_array = { global.GDP_list_array = THIS } }
					}
					add_to_array = { global.GDP_list_array = THIS }
				}
				calculate_GDP = no
				GDP_growth_calculation = yes
				gdp_growth_calc = yes
			}
			
		}
	}
	
	on_monthly = {
		effect = {
			every_country = {
				limit = {
					NOT = {
						has_country_flag = GDP_taobei_complete
					}
				}
				add_to_variable = { country_GDP = 1 }
				add_to_variable = { country_GDP_per_capita = 1 }
				calculate_gdp = yes
			}

			calculate_GDP = yes
			GDP_growth_calculation = yes
			force_update_map_mode = { mapmode = government_type_map_mode }
			force_update_map_mode = { mapmode = industrialisation_map_mode }
		}
	}
}
