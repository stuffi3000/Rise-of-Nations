		icon = GFX_decision_generic_form_nation
		allowed = {
			OR = {
				original_tag = AST
				original_tag = ENG
				original_tag = CAN
			}
		}
		available = {
			if = {
				limit = {
					has_game_rule = {
						rule = agw_greatwar_options
						option = enabled
					}
				}
				has_global_flag = interbellum
			}
			has_war = no
		}
		visible = {
			OR = {
				AND = {
					OR = {
						original_tag = ENG
						original_tag = CAN
					}
					has_subject = AST
					has_subject = NZL
				}
				AND = {
					original_tag = AST
					OR = {
						has_subject = NZL
						AND = {
							NOT = { country_exists = NZL }
							owns_state = 284
							owns_state = 723
						}
					}
				}
			}
			AST = {
				NOT = { has_cosmetic_tag = AST_australasia }
			}
		}
		fire_only_once = yes
		ai_will_do = {
			base = 0
			modifier = {
				add = 1000
				has_country_flag = has_british_monarchy
				has_government = neutrality
			}
			modifier = {
				add = 1000
				has_country_flag = has_british_monarchy
				has_government = neutrality
			}
		}
		cost = 150
		days_remove = 180
		cancel_trigger = {
			hidden_trigger = {
				OR = {
					has_war = yes
					if = {
						limit = {
							NOT = { original_tag = AST }
						}
						NOT = { has_subject = AST }
						NOT = { has_subject = NZL }
					}
					else = {
						AND = {
							NOT = { has_subject = NZL }
							NOT = {
								owns_state = 284
								owns_state = 723
							}
						}
					}
					AST = { has_cosmetic_tag = AST_australasia }
				}
			}
		}
		cancel_effect = {
			add_political_power = 100
		}
