
	Axis_Victory_IND_AFG = {
		cost = 100
		fire_only_once = yes
		days_remove = 7
		war_with_on_remove = AFG
		allowed = {
			original_tag = IND
		}
		visible = {
			has_global_flag = RON_Axis_Victory
		}
		available = {
			original_tag = IND
			country_exists = AFG
			NOT = {
				is_in_faction_with = AFG
			}
			NOT = {
				has_war_with = AFG
			}
			date > 1946.1.1
		}
		ai_will_do = {
			factor = 1000
			modifier = {
				factor = 0
				date < 1949.6.1
			}
			modifier = {
				#Need a land link
					NOT = {
						is_neighbor_of = AFG
					}
				factor = 0
			}
			modifier = {
				strength_ratio = {
					tag = AFG
					ratio < 1.5
				}
				factor = 0
			}
			modifier = {
				#Not worth the risk if we are losing the war
				IND = {
					surrender_progress > 0.01
				}
				factor = 0
			}
		}
		complete_effect = {
			add_threat = 2
		}
		remove_effect = {
			diplomatic_relation = { country = AFG relation = non_aggression_pact active = no }
					create_wargoal = {			expire = 100
				target = AFG
				type = annex_everything
			}
		}
	}
