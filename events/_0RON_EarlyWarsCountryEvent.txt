# War of the Golden Stool - This will be the template for rebelions and such, cause they don't have tags yet
country_event = {
	id = ron_early_wars.1
	title = ron_early_wars_news.1.t
	desc = ron_early_wars_news.1.d
	picture = GFX_report_event_generic_modern_coup

	fire_only_once = yes

	trigger = {
		original_tag = ENG
		date > 1900.3.1
		date < 1900.9.12
		274 = { is_controlled_by = ENG } # Ghana, where the rebellion happened
	}

	immediate = { # Lets Make it trigger the news version for the rest of the world
		hidden_effect = {
			every_other_country = {
				limit = { NOT = { tag = ENG } }
				news_event = { id = ron_early_wars_news.1 hours = 6 }
			}
		}
	}

	option = {
		name = ron_early_wars.x.c    # Execute Them
		ai_chance = { factor = 10 }
		if = { limit = { is_ai = no } add_political_power = -25 }
		random_list = {
			20 = { country_event = { id = ron_warsdiplo.1 hours = 1 } }
			40 = {  country_event = { id = ron_warsdiplo.2 hours = 1 } }
			40 = {  
				country_event = { id = ron_warsdiplo.3 hours = 1 } 
				add_to_temp_array = { rebel_state = 274 }
				create_rebellion = yes
			}
		}
	}
	option = {
		name = ron_early_wars.x.e     # Intimidate Them
		ai_chance = { factor = 10 }
		if = { limit = { is_ai = no } add_political_power = -50 }
		random_list = {
			40 = { country_event = { id = ron_warsdiplo.1 hours = 1 } }
			30 = { country_event = { id = ron_warsdiplo.2 hours = 1 } }
			30 = {
				country_event = { id = ron_warsdiplo.3 hours = 1 } 
				add_to_temp_array = { rebel_state = 274 }
				create_rebellion = yes
			}
		}
	}
	option = {
		name = ron_early_wars.x.f     # Reach a Compromise
		ai_chance = { factor = 10 }
		if = { limit = { is_ai = no } add_political_power = -100 }
		random_list = {
			60 = {country_event = { id = ron_warsdiplo.1 hours = 1 } }
			20 = { country_event = { id = ron_warsdiplo.2 hours = 1 } }
