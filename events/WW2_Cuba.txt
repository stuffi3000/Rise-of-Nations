	id = elections.2
	title = elections.2.t
	desc = elections.2.d
	picture = GFX_report_event_election_vote

	is_triggered_only = no
	fire_only_once = yes

	trigger = {
		tag = CUB
		date > 1940.5.1
		date < 1940.6.1
	}
	immediate = {
		create_country_leader = {
			name = "Fulgencio Batista"
			desc = "POLITICS_FULGENCIO_BATISTA_DESC"
			picture = "Fulgencio_Batista.dds"
			expire = "2060.1.1"
			ideology = despotism
		}
	}
	option = {
		name = elections.2.a
		ai_chance = { factor = 100 }
		hidden_effect = {
			country_event = { days = 1 id = news.299 }
		}
		add_political_power = 100
		set_politics = {
			ruling_party = neutrality
			elections_allowed = yes
		}
	}

	option = {
		name = elections.2.b
		ai_chance = { factor = 100 }
		add_political_power = -100
		add_popularity = {
			ideology = neutrality
			popularity = -0.1
		}
	}
}

add_namespace = haiti
country_event = {
	id = haiti.8
	title = haiti.8.t
	desc = haiti.8.d
	picture = GFX_report_event_albanian_king_zog

	is_triggered_only = yes

	option = { # Yield
		name = haiti.8.a
		ai_chance = {
			base = 40
			modifier = {
				factor = 0
				any_other_country = {
					is_major = yes
					NOT = { tag = CUB }
					OR = {
						has_guaranteed = HAI
						is_in_faction_with = HAI
					}
				}
			}
		}
		CUB = {
			country_event = { days = 1 id = haiti.6 }
		}
		custom_effect_tooltip = GAME_OVER_TT
	}
	option = { # Resist
		name = haiti.8.b
		ai_chance = {
			factor = 20
			modifier = {
				factor = 0
				OR = {
					AND = {
						is_in_faction_with = CUB
						ITA = { is_faction_leader = yes }
					}
					is_historical_focus_on = yes
					AND = {
						is_in_faction = no
						NOT = {
							any_other_country = {
								NOT = { tag = CUB }
								has_guaranteed = HAI
							}
						}
					}
				}
			}
		}
		CUB = {
			effect_tooltip = {
						create_wargoal = {			
				expire = 250
					type = take_state_focus
					target = HAI
					generator = { 318 }
				}
			}
			country_event = { days = 1 id = haiti.7 }
		}
	}
}
