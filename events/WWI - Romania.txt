# New Commie Gov
country_event = {
	id = ends_romania.4
	title = ends_romania.4.t
	desc = ends_romania.4.d
	picture = GFX_report_event_king_michael

	fire_only_once = yes

	trigger = {
		tag = ROM
		SOV = { controls_state = 79 }
		NOT = { has_global_flag = commie }
		date > 1943.1.1
		date < 1945.1.1
	}

	mean_time_to_happen = {
		days = 1
	}
	option = {
		name = ends_romania.4.a
		set_global_flag = commie
		set_global_flag = CMR
		news_event = { hours = 6 id = endsieg_news.124 }
		ROM = {
			add_ai_strategy = {
				type = alliance
				id = "SOV"
				value = 200
			}
		}
		GER = { remove_from_faction = ROM }

		white_peace = ENG
		white_peace = SOV
		white_peace = GER

		ROM = { declare_war_on = { target = GER type = annex_everything } }
		SOV = { add_to_faction = ROM }
		set_politics = {
			ruling_party = communism
			elections_allowed = no }

		set_popularities = {
			communism = 100
		}
		SOV = {
			puppet = ROM
			if = {
				limit = { 78 = { is_owned_by = ROM } }
				transfer_state = 78
			}
			if = {
				limit = { 80 = { is_owned_by = ROM } }
				transfer_state = 80
			}
		}
	}
immediate = { log = "Event: [ROOT.GetName] [GetDateString] file:WWI - Romania.txt line:193" }}
