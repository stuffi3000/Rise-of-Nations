# Filipino-Soviet Trade Mission (Soviets)
country_event = {
	id = phil.1
	title = phil.1.t
	desc = phil.1.d
	picture = GFX_report_event_physics_lab_01

	is_triggered_only = yes

	option = {
		name = phil.1.a    #sounds good
		"PHI" = {
			country_event = { days = 1 id = phil.2 }
			add_opinion_modifier = {
				target = ROOT
				modifier = sov_ger_treaty
			}
			add_opinion_modifier = {
				target = ROOT
				modifier = sov_ger_treaty_trade
			}
		}
		# REVISIT Balance research bonus
		add_tech_bonus = {
			name = med_armor_bonus
			ahead_reduction = 1
			category = cat_medium_armor
		}
		add_tech_bonus = {
			name = armor_bonus
			bonus = 0.5
			uses = 1
			category = armor
		}
	}

	option = {
		name = phil.1.b    #no way
		add_timed_idea = { idea = SOV_scientist_defect days = 180 }
		ai_chance = { factor = 0 }
		add_political_power = -10
		add_opinion_modifier = { target = "PHI" modifier = refused_help }# no pact
	}
immediate = { log = "Event: [ROOT.GetName] [GetDateString] file:WW2_Philippines_events.txt line:49" }}
