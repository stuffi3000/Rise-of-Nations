country_event = {
	id = mexico_1940.1
	title = mexico_1940.1.t
	desc = mexico_1940.1.d
	picture = GFX_report_event_leftist
	trigger = {
		date > 1939.1.1
		date < 1949.1.1
		original_tag = MEX
		NOT = {
			has_government = fascism
			has_government = communism
			has_government = democratic
			has_idea = Mugica
			has_idea = Maximino
			has_idea = padilla
		}
	}
	fire_only_once = yes

	option = {
		#Mugica
		name = mexico_1940.1.a
		ai_chance = {
			factor = 1
			modifier = {
				factor = 10
				communism > 0.40
			}
		}
		MEX = {
			country_event = { days = 1 id = mexico_1940.2 }
		}
		add_country_leader_role = {
			character = MEX_francisco_mugica
			promote_leader = yes
			country_leader = {
				desc = "Mugica_desc"
				expire = "2060.1.1"
				ideology = stalinism
			}
		}
		set_party_name = {
			ideology = communism
			long_name = "Cardenistas"
			name = "Cardenistas"
		}
		set_country_flag = election_1940
		set_country_flag = mugica
		set_country_flag = mexican_elections
	}

	option = {
		#Camacho
		name = mexico_1940.1.b
		ai_chance = {
			factor = 1
			modifier = {
				factor = 100
				OR = {
					communism < 0.40
					is_historical_focus_on = yes
				}
			}
		}
		MEX = {
			country_event = { days = 1 id = mexico_1940.2 }
		}
		add_country_leader_role = {
			character = MEX_manuel_avila_camacho
			promote_leader = yes
			country_leader = {
				desc = "manuel_desc"
				expire = "2060.1.1"
				ideology = militarism
			}
		}
		set_country_flag = election_1940
		set_country_flag = camacho
		set_country_flag = mexican_elections
	}
}
