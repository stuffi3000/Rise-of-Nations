# The Great Escape of The British Royals (After the Referendum)
country_event = {
	id = ww1_australia.13
	title = ww1_australia.12.t
	desc = ww1_australia.13.d
	picture = GFX_report_event_royal_parade
	
	fire_only_once = yes
	is_triggered_only = yes
	option = {
		name = ww1_australia.12.a
		ai_chance = {
			factor = 100
		}
		set_party_name = {
			ideology = neutrality
			long_name = ENG_neutrality_party_long
			name = ENG_neutrality_party
		}
		ENG = {
			ENG_george_v = {
				set_nationality = ROOT
				if = {
					limit = {
						has_ideology = semi_constitutional_monarchy
					}
					add_country_leader_trait = revolution_trauma
					promote_character = semi_constitutional_monarchy
				}
				else = {
					add_country_leader_role = {
						promote_leader = yes
						country_leader = {
							ideology = semi_constitutional_monarchy
							traits = { revolution_trauma }
						}
					}
				}
			}
		}
		add_ideas = george_v
		if = {
			limit = {
				has_idea = AST_idea_Dominion_Army
			}
			remove_ideas = AST_idea_Dominion_Army
		}

		if = {
			limit = {
				has_idea = AST_idea_Citizen_Army
			}
			remove_ideas = AST_idea_Citizen_Army
		}
		if = {
			limit = {
				ENG = {
					has_idea = ENG_Staff
				}
			}
			add_ideas = ENG_Staff
		}

		add_popularity = {
			ideology = neutrality
			popularity = 0.1
		}

		add_stability = -0.03

		hidden_effect = {
			517 = {
				add_building_construction= {
					type = air_base
					level = 1
					instant_build = yes
				}
			}
		}
		add_ideas = ENG_imperial_refuge
		set_country_flag = has_british_monarchy

		if = {
			limit = {
				NZL = {
					is_subject = no
					has_government = democratic
				}
			}
			set_autonomy = {
				target = NZL
				autonomous_state = autonomy_dominion
				freedom_level = 0.5
			}

			NZL = {
				set_politics = {
					ruling_party = democratic
					elections_allowed = yes
				}
				set_popularities = {
					democratic = 90
					fascism = 0
					communism = 1
					neutrality = 9
				}
			}
		}
	}
	option = {
		name = ww1_australia.12.b
		ai_chance = {
			base = 25
		}

		add_popularity = {
			ideology = neutrality
			popularity = 0.03
		}

		add_stability = -0.05
	}
}
