# Persian coup d'etat 1921
country_event = {
	id = persia.1
	title = peria.1.t
	desc = persia.1.d
	picture = GFX_report_event_persiancoup

	fire_only_once = yes

	trigger = {
		NOT = { date > 1932.12.1 }
		tag = PER
		date > 1921.2.18
	}

	mean_time_to_happen = {
		days = 1
	}

	option = {
		name = persia.1.a
		set_global_flag = persiancoup
		add_stability = 0.05
		# Qajar Government Falls Pahlavi takes over
		retire_country_leader = yes
		create_country_leader = {
			name = "Reza Shah Pahlavi"
			desc = "POLITICS_REZA_SHAH_DESC"
			picture = "Portrait_Persia_Reza_Shah.dds"
			expire = "1941.9.16"
			ideology = paternal_autocrat
			traits = {
				#
			}
		}
	}
}
