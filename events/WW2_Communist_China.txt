					is_ai = yes

			}
			AND = {
				date > 1945.10.15
				has_war = no
				PRC = {
					is_ai = no
				}
			}
			AND = {
				date > 1948.5.1
				has_war = yes
				PRC = {
					is_ai = no
				}
			}
		}
		tag = SOV
		NOT = { has_country_flag = counter_purge }
		has_government = communism
		has_start_date < 1960.1.1
		date < 2000.1.1
	}

	mean_time_to_happen = { days = 1 }

		hidden_effect = {
			kill_country_leader = yes
