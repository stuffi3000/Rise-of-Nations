### Manuel A. Rodriguez isn't feeling well.

country_event = {
	id = raother.1
	title = raother.1.t
	desc = raother.1.d

	picture = GFX_raothers_1

	fire_only_once = yes

	trigger = {
		TAG = ARG
		date > 1936.2.1
		date < 1949.1.1
	}

	option = {
		name = raother.1.a

		hidden_effect = {
			set_country_flag = ra_arg_rodriguez_dies
			country_event = { days = 7 id = raother.2 }
		}

	}
	option = {
		name = raother.1.b

		hidden_effect = {
			set_country_flag = ra_arg_rodriguez_lives
			country_event = { days = 7 id = raother.2 }
		}
	}
