# Panic of 1901

	id = ron_economicnews.1
	title = ron_economicnews.1.t
	desc = ron_economicnews.1.d
	picture = GFX_news_event_panic_of_1901

	major = yes
	fire_only_once = yes

	trigger = {
		tag = USA
		NOT = { has_global_flag = panic_of_1901 }

	}

		set_global_flag = panic_of_1901
	}



	option = {
		name = ron_economicnews.1.a
	}
