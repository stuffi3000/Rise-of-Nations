# Death of Menelik II
country_event = {
	id = ww1_ethiopia.1
	title = ww1_ethiopia.1.t
	desc = ww1_ethiopia.1.d
	picture = GFX_report_event_menelik

	is_triggered_only = yes

	trigger = {
		date > 1913.12.12
		date < 1913.12.14
	}

	option = {
		name = ww1_ethiopia.1.a
		effect_tooltip = {
			kill_country_leader = yes
		}
		hidden_effect = {
			retire_character = ETH_menelik_ii
		}
		promote_character = ETH_lij_iyasu
	}
}
