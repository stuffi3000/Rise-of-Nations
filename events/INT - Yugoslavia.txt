add_namespace = yugoslavia

# Alexander Assassinated
country_event = {
	id = yugoslavia.1
	title = yugoslavia.1.t
	desc = yugoslavia.1.d
	picture = GFX_report_event_alexander_dead_yug

	fire_only_once = yes

	trigger = {
		NOT = { date > 1935.1.1 }
		tag = YUG
		date > 1934.10.9
	}

	mean_time_to_happen = { days = 1 }

	immediate = {
		set_global_flag = alexander_yugo_dead
	}
	option = {
		name = yugoslavia.1.a
		add_stability = -0.05
		set_politics = {
			ruling_party = paternal_autocrat
			elections_allowed = no    }

        set_popularities = {
            populist = 0
            fascism = 0
            paternal_autocrat = 45
            social_conservative = 35
            market_liberal = 5
            social_liberal = 5
            social_democrat = 0
            leninist = 0
            stalinist = 0
            anarchist = 10
        }
		# Alexander I Dies
		kill_country_leader = yes
		create_country_leader = {
			name = "Prince Regent Paul"
			desc = "POLITICS_REGENT_PAUL_DESC"
			picture = "Portrait_Yugoslavia_Paul_Regent.dds"
			expire = "1941.3.27"
			ideology = paternal_autocrat
			traits = {
				#
			}
		}
	}
}
