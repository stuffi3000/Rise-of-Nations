country_event = {
	id = bs.17
	title = bs.17.t
	desc = bs.17.d
	picture = GFX_report_event_generic_sign_treaty2

	fire_only_once = yes

	trigger = {
		#Fires for one of the new German states to transfer old German territory.
		country_exists = KOR
		OR = {
			country_exists = ROK
			country_exists = DPK
		}
		OR = {
			tag = ROK
			tag = DPK
		}
	}



	option = {
		name = bs.17.a
		if = {
			limit = {
				country_exists = ROK
			}
			#Give ROK their cores
			every_state = {
				limit = {
					is_owned_and_controlled_by = KOR
					is_core_of = ROK
				}
				ROK = {
					transfer_state = PREV
				}
			}
		}
		if = {
			limit = {
				country_exists = ROK
			}
			#Give ROK their cores
			every_state = {
				limit = {
					is_owned_and_controlled_by = JAP
					is_core_of = ROK
				}
				ROK = {
					transfer_state = PREV
				}
			}
		}
		if = {
			limit = {
				country_exists = DPK
			}
			#Give East Germany their cores
			every_state = {
				limit = {
					is_owned_and_controlled_by = KOR
					is_core_of = DPK
				}
				DPK = {
					transfer_state = PREV
				}
			}
		}
		if = {
			limit = {
				country_exists = DPK
			}
			#Give East Germany their cores
			every_state = {
				limit = {
					is_owned_and_controlled_by = JAP
					is_core_of = DPK
				}
				DPK = {
					transfer_state = PREV
				}
			}
		}
		if = {
			limit = {
				country_exists = JAP
			}

			every_state = {
				limit = {
					is_owned_and_controlled_by = JAP
					is_claimed_by = USA
				}
				USA = {
					transfer_state = PREV
				}
				remove_core_of = JAP
			}
		}
		if = {
			limit = {
				country_exists = JAP
			}

			every_state = {
				limit = {
					is_owned_and_controlled_by = JAP
					is_claimed_by = SOV
				}
				SOV = {
					transfer_state = PREV
				}
				remove_core_of = JAP
			}
		}
		if = {
			limit = {
				DPK = {
					KOR = {
						has_government = PREV
					}
				}
			}
			#If KOR has changed government, remaining territory is passed to the corresponding new state (ROK or DPK)
			every_state = {
				limit = {
					is_owned_and_controlled_by = KOR
					NOT = {
						OR = {
							is_core_of = DPK
							is_core_of = ROK
						}
					}
				}
				DPK = {
					transfer_state = PREV
				}
			}
		}
		if = {
			limit = {
				ROK = {
					KOR = {
						has_government = PREV
					}
				}
			}
			#See above
			every_state = {
				limit = {
					is_owned_and_controlled_by = KOR
					NOT = {
						OR = {
							is_core_of = DPK
							is_core_of = ROK
						}
					}
				}
				ROK = {
					transfer_state = PREV
				}
			}
		}
		if = {
			limit = {
				any_state = {
					is_owned_and_controlled_by = KOR
					NOT = {
						OR = {
							is_core_of = ROK
							is_core_of = DPK
						}
					}
				}
			}
			#Any remaining cores go to ROK, because democracy is fun
			every_state = {
				limit = {
					is_owned_and_controlled_by = KOR
					NOT = {
						OR = {
							is_core_of = ROK
							is_core_of = DPK
						}
					}
				}
				ROK = {
					transfer_state = PREV
				}
			}
		}
		if = {
			limit = {
				country_exists = ROK
				country_exists = DPK
			}
			#Remove KOR cores
			hidden_effect = {
				every_state = {
					limit = {
						is_core_of = KOR
					}
					remove_core_of = KOR
				}
			}
		}
	}
}
