	}

	option = {

		}
		set_autonomy = { target = LAT autonomy_state = autonomy_dominion }
		if = {
			limit = {
				is_in_faction = yes
			}
			random_country = {
				limit = {
					is_in_faction_with = LIT

				}
				add_to_faction = LAT
			}
		}
	}

}
