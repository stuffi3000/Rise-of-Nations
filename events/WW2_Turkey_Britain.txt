		}

				limit = {

		name = tur_britain.100.b

					}
				}
				NOT = {
					any_other_country = {
						has_war_with = TUR
					}
				}

					any_other_country = {
						NOT = {

						}

						}

							is_in_faction_with = TUR
						}
					}
				}
			}

				factor = 0

					is_in_faction_with = TUR

						is_puppet_of = TUR
