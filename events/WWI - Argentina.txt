add_namespace = argentina

# Tragic Week
country_event = {
	id = argentina.1
	title = argentina.1.t
	desc = argentina.1.d
	picture = GFX_report_event_tragic_week

	fire_only_once = yes

	trigger = {
		tag = ARG
		date > 1919.1.7
		date < 1919.1.10
	}

	mean_time_to_happen = { days = 1 }

	immediate = {
		set_global_flag = tragic_week
	}
	option = {
		name = argentina.1.a
		add_stability = -0.05
		add_political_power = -50
		hidden_effect = {
			news_event = { hours = 6 id = endsieg_news.42 }
		}
	}
}
