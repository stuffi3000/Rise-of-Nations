#Second Civil War Begins
country_event = {
	id = RON_Russian_Alt_History.1
	title = RON_Russian_Alt_History.1.t
	desc = RON_Russian_Alt_History.1.d
	picture = GFX_report_event_RON_Russian_Alt_History_White_Army
	is_triggered_only = yes
	option = {
		name = RON_Russian_Alt_History.1.a
		ai_chance = {
			factor = 10
		}
		
		if = {
			limit = {
				SOV = { is_ai = yes }
			}
			start_civil_war = {
				ruling_party = neutrality
				ideology = communism
				size = 0.4
				capital = 571
				states = { 571 516 644 637 579 580 653 572 1000 402 587 583 403 590 562 1090 560 970 655 576 568 584 585 405 742 732 574 575 561 657 408 409 567 566 564 565 563 567 404 586 588 589 654 569 570 40 577 578 }
			}
		}
		
		if = {
			limit = {
				SOV = { is_ai = no }
			}
			start_civil_war = {
				ruling_party = neutrality
				ideology = communism
				size = 0.5
				capital = 571
				states = { 571 516 644 637 579 580 653 572 1000 402 587 583 403 590 562 1090 560 970 655 576 568 584 585 405 742 732 574 575 561 657 408 409 567 566 564 565 563 567 404 586 588 589 654 569 570 40 577 578 }
			}
		}
		
		hidden_effect = {
			RUC = {
				RON_increase_conscription_law = yes
				RON_increase_economy_law = yes
			}
		}
		if = {
			limit = {
				has_country_flag = RON_Stalin_Escapes
			}
		RUC = {
			create_country_leader = {
				name = "Joseph Stalin"
				desc = "POLITICS_JOSEPH_STALIN_DESC"
				picture = "gfx/leaders/SOV/Portrait_Soviet_Joseph_Stalin.dds"
				ideology = stalinism
				traits = {
					man_of_steel
				}
			}
		}
		}
		if = {
			limit = {
				has_country_flag = RON_Stalin_Killed
			}
			RUC = {
				retire_ideology_leader = communism
				create_country_leader = {
					name = "Lavrentiy Beria"
					picture = "gfx/leaders/SOV/Portrait_SOV_Beria.dds"
					ideology = stalinism
				}
			}
		}
		RON_increase_conscription_law = yes
		RON_increase_economy_law = yes
		
		set_global_flag = RON_Second_Russian_Civil_War
		
		news_event = {
			id = RON_Russian_Alt_History_News.1
			hours = 6
		}
			
	}
}
