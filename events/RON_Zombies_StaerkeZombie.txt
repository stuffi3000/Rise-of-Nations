add_namespace = StaerkeZombie


country_event = {
	id = StaerkeZombie.4
	title = "StaerkeZombie.4.name"
	desc = "StaerkeZombie.4.desc"
	fire_only_once = yes
	hidden = yes

	trigger = {
		has_global_flag = ZOMBIE_MODE
		AND = {
			has_global_flag = INSANE
			has_global_flag = zom_india
		}
	}
	mean_time_to_happen = { days = 10 }
	option = {
		name = StaerkeZombie.1.ab
		ZOM = {
			add_manpower = 6000000
			load_oob = "ZOM_staerke1_india"
			add_ideas = {
				Besser
			}
		}
		set_division_template_lock = {
			division_template = "Zombie Horde"
			is_locked = yes
		}
	}
}
