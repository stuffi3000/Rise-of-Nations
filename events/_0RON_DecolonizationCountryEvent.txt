# Panama Independence from Colombia 1903
country_event = {
	id = ron_decolonizationcountry.1
	title = ron_decolonizationcountry.1.t
	desc = ron_decolonizationcountry.1.d
	picture = GFX_report_event_generic_cabinet_meeting
	fire_only_once = yes
	is_triggered_only = yes
	trigger = {
		original_tag = COL
		date > 1903.10.3
		date < 1903.11.3
		OR = {
			owns_state = 304
			PAN = {
				is_subject_of = COL
			}
		}
	}
	immediate = {
		hidden_effect = {

			COL = {
				release = PAN
				end_puppet = PAN
			}
			PAN = {
				news_event = {
					id = ron_decolonizationnews.1
					hours = 6
				}
			}
		}
	}

	option = {
		name = ron_decolonizationcountry.x.a
		ai_chance = {
			factor = 10
		}
		PAN = {
			country_event = {
				id = ron_decolonizationdiplo.1
				hours = 1
			}
		}
	}
	option = {
		name = ron_decolonizationcountry.x.b
		ai_chance = {
			factor = 10
			modifier = {
				factor = 0
				is_historical_focus_on = yes
			}
		}
		PAN = {
			country_event = {
				id = ron_decolonizationdiplo.2
				hours = 1
			}
		}
	}
}

# Bulgarian Declaration of Independence 1908
country_event = {
	id = ron_decolonizationcountry.3
	title = ron_decolonizationcountry.3.t
	desc = ron_decolonizationcountry.3.d
	picture = GFX_report_event_DNC_Expansion_country227
	fire_only_once = yes
	trigger = {
		original_tag = OTT
		date > 1908.10.5
		date < 1908.11.5
		OR = {
			any_state = {
				is_owned_by = ROOT
				is_core_of = BUL
			}
			BUL = {
				is_subject_of = OTT
			}
		}
	}
	immediate = {
		hidden_effect = {
			OTT = {
				release = BUL
				end_puppet = BUL
			}
			BUL = {
				news_event = {
					id = DNC_Expansion_news.227
					hours = 6
				}
				every_state = {
					limit = {
						is_core_of = BUL
						is_owned_by = OTT
					}
					BUL = {
						transfer_state = PREV
					}
				}
			}
		}
	}

	option = {
		name = ron_decolonizationcountry.x.a
		ai_chance = {
			factor = 50
		}
		BUL = {
			country_event = {
				id = ron_decolonizationdiplo.1
				hours = 1
			}
		}
	}
	option = {
		name = ron_decolonizationcountry.x.b

		trigger = {
			has_civil_war = no
			NOT = {
				has_country_flag = DNC_youngturklost
			}
		}

		ai_chance = {
			factor = 10
			modifier = {
				factor = 0
				is_historical_focus_on = yes
			}
		}
		BUL = {
			country_event = {
				id = ron_decolonizationdiplo.2
				hours = 1
			}
		}
	}
}

	id = ron_decolonizationcountry.106
	title = ron_decolonizationcountry.106.t
	desc = ron_decolonizationcountry.106.d
	picture = GFX_report_event_generic_cabinet_meeting
	fire_only_once = yes
	trigger = {
		original_tag = PER
		date > 1979.4.1
		date < 1979.5.1
		is_subject = no
		NOT = {
			has_government = islamism
		}
		country_exists = PER
	}
	option = {
		name = ron_decolonizationcountry.106.a
		ai_chance = {
			factor = 10
		}
		set_politics = {
			ruling_party = islamism
			last_election = "1980.1.1"
			election_frequency = 48
			elections_allowed = no
		}
		set_popularities = {
			islamism = 100
		}
		news_event = {
			id = ron_coupnews.3
			hours = 6
		}
	}
	option = {
		name = ron_decolonizationcountry.106.b
		ai_chance = {
			factor = 10
			modifier = {
				factor = 0
				is_historical_focus_on = yes
			}
		}
		random_list = {
			50 = {
				start_civil_war = {
					ruling_party = ROOT

			}
			50 = {
				news_event = {
					id = ron_coupnews.6

				}
			}
		}
	}
}
