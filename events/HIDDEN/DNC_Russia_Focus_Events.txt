# question of mongolia

country_event = {
	id = dncrussia.1
	title = dncrussia.1.t
	desc = dncrussia.1.d
	#picture = 
	is_triggered_only = yes
	fire_only_once = yes
	
	option = { # mongolia puppet
		name = dncrussia.1.a
		ai_chance = {
			factor = 90
		}
		if = {
			limit = { has_dlc = "Together for Victory" }
			RUS = {
				set_autonomy = {
					target = MON
					autonomous_state = autonomy_integrated_puppet
				}
			}
			else = {
				RUS = { puppet = MON }
			}
		}
		country_event = { id = dncrussia.1000 hours = 12 } #mongolia submits
	}
	option = { # mongolia annexed
		name = dncrussia.1.b
		ai_chance = {
			factor = 9
			modifier = {
				factor = 0
				is_historical_focus_on = yes
			}
		}
		any_owned_state = { add_core_of = RUS }
		RUS = {
			annex_country = { target = MON }
			country_event = { id = dncrussia.1000 hours = 12 } #mongolia submits
		}
	}
	option = { # mongolia refuses
		name = dncrussia.1.c
		ai_chance = {
			factor = 1
			modifier = {
				factor = 0
				is_historical_focus_on = yes
			}
		}
		RUS = {
			country_event = { id = dncrussia.1001 hours = 12 } #mongolia refuses
		}
	}
}
