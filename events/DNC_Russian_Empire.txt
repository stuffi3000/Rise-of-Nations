			modifier = {
				has_government = democratic

			}
		}
		add_political_power = -75
		add_war_support = -0.1
		add_stability = -0.1
		every_state = {
			limit = {
				is_core_of = FIN
				NOT = {
					is_controlled_by = FROM
				}
			}
			force_enable_resistance = RUE
			add_resistance = 10
		}
	}
	
	option = {
		name = dnc_russian_empire.1.b
		ai_chance = {
			base = 10
			modifier = {
				has_political_power < 50
				add = 25
			}
			modifier = {
				has_stability < 0.5
				add = 25
			}
		}
		random_list = {
			60 = {
				add_war_support = -0.01
				every_state = {
					limit = {
						is_core_of = FIN
					}
					remove_core_of = RUE
				}
				hidden_effect = {
					FROM = {
						country_event = {
							id = dnc_germany.29 #telling FROM they succeded
						}
					}
				}
			}
			40 = {
				add_political_power = 100
				add_war_support = 0.1
				add_stability = 0.1
				FROM = {
