# question of mongolia

country_event = {
	id = ww1russia.1
	title = ww1russia.1.t
	desc = ww1russia.1.d
	#picture = 
	is_triggered_only = yes
	fire_only_once = yes

	option = { # mongolia puppet
		name = ww1russia.1.a
		ai_chance = {
			factor = 90
		}
		if = {
			limit = { has_dlc = "Together for Victory" }
			RUS = {
				set_autonomy = {
					target = MON
					autonomous_state = autonomy_integrated_puppet
				}
			}
			RUS = {
				set_autonomy = {
					target = TAN
					autonomous_state = autonomy_integrated_puppet
				}
			}
			else = {
				RUS = { puppet = MON }
				RUS = { puppet = TAN }
			}
		}
		country_event = { id = ww1russia.1000 hours = 12 } #mongolia submits
	}
	option = { # mongolia annexed
		name = ww1russia.1.b
		ai_chance = {
			factor = 9
			modifier = {
				factor = 0
				is_historical_focus_on = yes
			}
		}
		# any_owned_state = { add_core_of = RUS } 
		RUS = {
			annex_country = { target = MON }
			annex_country = { target = TAN }
			country_event = { id = ww1russia.1000 hours = 12 } #mongolia submits
		}
	}
	option = { # mongolia refuses
		name = ww1russia.1.c
		ai_chance = {
			factor = 1
			modifier = {
				factor = 0
				is_historical_focus_on = yes
			}
		}
		RUS = {
			country_event = { id = ww1russia.1001 hours = 12 } #mongolia refuses
		}
	}
}
